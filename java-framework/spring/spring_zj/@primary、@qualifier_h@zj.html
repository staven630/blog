# @Primary

вҖғвҖғ@Autowired й»ҳи®Өж №жҚ®зұ»жқҘжҹҘжүҫе’ҢжіЁе…Ҙе®№еҷЁдёӯзҡ„еҜ№иұЎпјҢеҰӮжһңеӯҳеңЁеҗҢдёҖдёӘзұ»зҡ„еӨҡдёӘ Bean е®һдҫӢиў«е®№еҷЁз®ЎзҗҶзҡ„зҠ¶еҶөпјҢеңЁдҪҝз”Ё@Autowired иЈ…й…ҚиҜҘзұ»зҡ„дҫқиө–еҜ№иұЎж—¶дјҡжҠҘ UnsatisfiedDependencyException зҡ„ејӮеёёпјҢжҸҗзӨә expected single matching bean but found X,е®№еҷЁеҲқе§ӢеҢ–еӨұиҙҘгҖӮеҸҜд»ҘеңЁиҜҘзұ»зҡ„жҹҗдёӘ Bean зҡ„й…ҚзҪ®дёӯи®ҫзҪ®иҜҘ Bean дҪңдёәдҫқиө–жіЁе…Ҙзҡ„дё»еҖҷйҖүйЎ№и§ЈеҶіжӯӨй—®йўҳпјҢеҜ№еә” XML й…ҚзҪ®е’Ң Java жіЁи§Јй…ҚзҪ®зҡ„ж–№ејҸеҲҶеҲ«дёәпјҡ

  • еңЁ XML зҡ„<bean>й…ҚзҪ®дёӯдҪҝз”Ё primary еұһжҖ§и®ҫзҪ®жҳҜеҗҰжҳҜдё»еҖҷйҖү BeanпјҢtrue иЎЁзӨәдјҳе…ҲдҪҝз”ЁгҖӮ
  • еңЁдҪҝз”Ё Java жіЁи§ЈиҝӣиЎҢз»„д»¶е’Ңдҫқиө–й…ҚзҪ®зҡ„ж–№ејҸдёӢпјҢеҸҜд»ҘдҪҝз”Ё@Primary жіЁи§ЈеңЁзұ»е’Ңж–№жі•дёҠгҖӮ

вҖғвҖғ@Primary еҸҜд»ҘдҪҝз”ЁеңЁ@Component жіЁи§Јзҡ„зұ»дёӯпјҢд№ҹеҸҜд»ҘдҪҝз”ЁеңЁ@Bean жіЁи§Јзҡ„ж–№жі•дёҠгҖӮ

  • дҪҝз”ЁеңЁ@Component жіЁи§Јзұ»зҡ„еңәжҷҜпјҡA жҳҜдёҖдёӘжҺҘеҸЈпјҢBAгҖҒCA жҳҜ A жҺҘеҸЈзҡ„е®һзҺ°зұ»пјҢе№¶дё”иҝҷдёӨдёӘе®һзҺ°зұ»дҪҝз”Ё@Component жіЁи§Јдёәз»„д»¶гҖӮеҰӮжһңйҖҡиҝҮ@Autowired жіЁе…Ҙ A жҺҘеҸЈзұ»еһӢзҡ„дҫқиө–пјҢдјҡжүҫеҲ°дёӨдёӘ BeanпјҢжӯӨж—¶еҸҜд»ҘеңЁ BA жҲ– CA д№ӢдёҖдёӯдҪҝз”Ё@Primary жіЁи§ЈгҖӮ
  • еңЁж–№жі•дёӯдҪҝз”Ё@Bean жіЁи§ЈпјҢеҸҜд»ҘиҫҫжҲҗеҜ№еҗҢдёҖдёӘзұ»зҡ„еӨҡдёӘ Bean е®һдҫӢзҡ„жіЁеҶҢпјҢиҝҷд№ҹжҳҜ@Primary жӣҙдёәеёёи§Ғзҡ„дҪҝз”Ёж–№ејҸгҖӮ
@Bean // з»„д»¶жіЁи§Ј
@Primary
public void getUserInfo(User user) {
   return new User('staven');
}

# @Qualifier

вҖғвҖғйҷӨдәҶдҪҝз”Ё@Primary жҢҮе®ҡдё»еҖҷйҖүд№ӢеӨ–пјҢиҝҳеҸҜд»ҘдҪҝз”Ё@QualifierпјҢе…¶е’Ң@Autowired ж №жҚ® Bean зҡ„еҗҚеӯ—жқҘжҹҘжүҫдҫқиө–еҜ№иұЎпјҢиҝӣиЎҢз»ҶзІ’еәҰзҡ„й…ҚзҪ®гҖӮ
2/211
вҖғвҖғ@Qualifier еҸҜд»ҘдҪҝз”ЁеңЁеұһжҖ§гҖҒж–№жі•е’ҢеҸӮж•°дёӯгҖӮ

@Autowired
@Qualifier("oneUser") // @Qualifier(value="oneUser")
public void getUserInfo(User user) {
   return new User('staven');
}
